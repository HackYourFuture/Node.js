add : usage: -a <string >
       Description : adds the specified string to a file 
         
remove : usage: -r <lineNumber> 
       Description : removes the specified  line number from  a file 

update : usage: -u <lineNumber> <update todo>
         Description : Updates todo on the lineNumber  

reset : usage : -R 
         Description : resets a file to empty

list : usage : -l
        Description : list of todos
